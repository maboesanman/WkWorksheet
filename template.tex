\newcommand{\strokeOrder}[3]{
  \IfFileExists{out/#1-0#2.svg}{}{

  }
  \includesvg[width=#3,height=#3,keepaspectratio]{out/#1-0#2.svg}
}

% \KanjiGrid{<cell size mm>}{<rows>}
\newcommand{\KanjiGrid}[2]{%
  \begin{tikzpicture}[x=-#1, y=-#1, line join=round, line cap=round]
    % Convert arguments to numeric TikZ variables
    \pgfmathsetmacro{\cellsize}{#1}
    \pgfmathtruncatemacro{\rows}{floor(\textheight / #1)}

    \draw[black] (0, 0) rectangle (1, 1);
    \draw[lightgray] (0, 0.5) -- (1, 0.5);
    \draw[lightgray] (0.5, 0) -- (0.5, 1);
    \node[anchor=center] at (0.5, 0.5){
      \KanjiFont
      {\fontsize{\fpeval{0.88*(#1/1pt)}}{\fpeval{0.88*1.2*(#1/1pt)}}\selectfont #2}
    };


    \draw[black] (0, 1) rectangle (1, 3);
    \draw[lightgray] (0, 1.5) -- (1, 1.5);
    \draw[lightgray] (0, 2.0) -- (1, 2.0);
    \draw[lightgray] (0, 2.5) -- (1, 2.5);
    \draw[lightgray] (0.5, 1) -- (0.5, 3);
    \node[anchor=center] at (0.25, 1.25){
      \strokeOrder{#2}{1}{\dimexpr#1/2\relax}
    };
    \node[anchor=center] at (0.25, 1.75){
      \strokeOrder{#2}{2}{\dimexpr#1/2\relax}
    };
    \node[anchor=center] at (0.25, 2.25){
      \strokeOrder{#2}{3}{\dimexpr#1/2\relax}
    };
    \node[anchor=center] at (0.25, 2.75){
      \strokeOrder{#2}{4}{\dimexpr#1/2\relax}
    };
    \node[anchor=center] at (0.75, 1.25){
      \strokeOrder{#2}{5}{\dimexpr#1/2\relax}
    };
    \node[anchor=center] at (0.75, 1.75){
      \strokeOrder{#2}{6}{\dimexpr#1/2\relax}
    };
    \node[anchor=center] at (0.75, 2.25){
      \strokeOrder{#2}{7}{\dimexpr#1/2\relax}
    };
    \node[anchor=center] at (0.75, 2.75){
      \strokeOrder{#2}{8}{\dimexpr#1/2\relax}
    };

    


    % boxes:
    \foreach \y in {4,...,\rows} {
      \draw[lightgray] (0, \y-0.5) -- (1, \y-0.5);
      \draw[lightgray] (0.5, \y-1) -- (0.5, \y);
      \draw[black] (0, \y-1) rectangle (1, \y);
    }

    \foreach \y in {4,...,6} {
      \node[anchor=center, text=gray!25] at (0.5, \y-0.5){
        \KanjiFont
        {\fontsize{\fpeval{0.88*(#1/1pt)}}{\fpeval{0.88*1.2*(#1/1pt)}}\selectfont #2}
      };
    }
  \end{tikzpicture}%
}
