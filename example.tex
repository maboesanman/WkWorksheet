\newcommand{\KanjiGrid}[1]{%
  \begin{center}%
  \begin{tikzpicture}[x=\gridsize, y=-\gridsize, line join=round, line cap=round]
  % Convert arguments to numeric TikZ variables
  \pgfmathtruncatemacro{\rows}{\gridheight}
  \pgfmathtruncatemacro{\cols}{\gridwidth}

  \pgfmathsetmacro{\kanjicount}{0}
  % once, in your preamble:
  \def\kanjicount{0}
  \foreach [count=\i] \kanji in {#1} { \xdef\kanjicount{\i} }


  \def\offset{0}% Set offset depending on page parity
  \ifodd\value{page}
    \xdef\offset{\kanjicount}%
  \else
    \xdef\offset{\gridwidth}%
  \fi
  

  \path[use as bounding box] (0, 0) rectangle (\cols, \rows);

  \foreach [count=\i] \kanji in {#1} {
    \begin{scope}[xshift=\gridsize*(\i - 1 + \offset - \kanjicount)]
    \draw[lightgray] (0, 0.5) -- (1, 0.5);
    \draw[lightgray] (0.5, 0) -- (0.5, 1);
    \draw[black] (0, 0) rectangle (1, 1);
    \node[anchor=center] at (0.5, 0.5){
      \Kaisho
      {\fontsize{\fpeval{0.88*(\gridsize/1pt)}}{\fpeval{0.88*1.2*(\gridsize/1pt)}}\selectfont \kanji}
    };


    \draw[lightgray] (0, 1.5) -- (1, 1.5);
    \draw[lightgray] (0, 2.0) -- (1, 2.0);
    \draw[lightgray] (0, 2.5) -- (1, 2.5);
    \draw[lightgray] (0.5, 1) -- (0.5, 3);
    \draw[black] (0, 1) rectangle (1, 3);
    \node[anchor=center] at (0.25, 1.25){
      \strokeOrder{\kanji}{1}{\dimexpr\gridsize/2\relax}
    };
    \node[anchor=center] at (0.25, 1.75){
      \strokeOrder{\kanji}{2}{\dimexpr\gridsize/2\relax}
    };
    \node[anchor=center] at (0.25, 2.25){
      \strokeOrder{\kanji}{3}{\dimexpr\gridsize/2\relax}
    };
    \node[anchor=center] at (0.25, 2.75){
      \strokeOrder{\kanji}{4}{\dimexpr\gridsize/2\relax}
    };
    \node[anchor=center] at (0.75, 1.25){
      \strokeOrder{\kanji}{5}{\dimexpr\gridsize/2\relax}
    };
    \node[anchor=center] at (0.75, 1.75){
      \strokeOrder{\kanji}{6}{\dimexpr\gridsize/2\relax}
    };
    \node[anchor=center] at (0.75, 2.25){
      \strokeOrder{\kanji}{7}{\dimexpr\gridsize/2\relax}
    };
    \node[anchor=center] at (0.75, 2.75){
      \strokeOrder{\kanji}{8}{\dimexpr\gridsize/2\relax}
    };

    \foreach \y in {4,...,\rows} {
      \draw[lightgray] (0, \y-0.5) -- (1, \y-0.5);
      \draw[lightgray] (0.5, \y-1) -- (0.5, \y);
      \draw[black] (0, \y-1) rectangle (1, \y);
    }

    \foreach \y in {4,...,6} {
      \node[anchor=center, text=gray!25] at (0.5, \y-0.5){
        \Kaisho
        {\fontsize{\fpeval{0.88*(\gridsize/1pt)}}{\fpeval{0.88*1.2*(\gridsize/1pt)}}\selectfont \kanji}
      };
    }

    \end{scope}%
  }
  \end{tikzpicture}%
\end{center}%
}

    \ifnum\i<#1

      \null
      \newcolumn
      \fi
      \advance\i by 1
  \repeat
}{%
  \end{multicols*}
}
